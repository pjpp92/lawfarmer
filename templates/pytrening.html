Jak działa request?

Stwórzcie plik z rozszerzeniem .py a następnie wklejcie do niego poniższy kod:

import os

from flask import Flask
from flask.ext.sqlalchemy import SQLAlchemy

basedir = os.path.abspath(os.path.dirname(__file__))

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///' + os.path.join(basedir, 'app.db')
db = SQLAlchemy(app)

class Director(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(80))
    year_of_birth = db.Column(db.Integer)
    alive = db.Column(db.Boolean(create_constraint=True, name=None))

    def __repr__(self):
        return '<Director: %s>' % self.name

class Movie(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(80), unique=True)
    type = db.Column(db.String(80))
    director_id = db.Column(db.Integer, db.ForeignKey(Director.id), primary_key=True)
    year_created = db.Column(db.Integer)
    director = db.relationship('Director', foreign_keys='Movie.director_id')

    def __repr__(self):
        return '<Movie: %s>' % self.title
